Para iniciar um projeto com NextJs
pode ser pelo npx ou yarn a diferença é que yarn é mais rápida a instalação.
 
Tem que instalar o instalador yarn antes com
npm install --global yarn

O comando Shift + Alt + F

--------------------------
Verifique:
yarn --version


Formas de começar. 
yarn create-next-app NOME-DO-PROJETO --typescript

OU
 yarn create next-app .    Com ponto para usar a propria pasta que acessou como nome do projeto

yarn add typescript -D

-----------------Criar src/components/ActiveLink/index.tsx:


import { ReactElement, cloneElement } from 'react';
import { useRouter } from 'next/router';
import Link, { LinkProps } from 'next/link';

interface ActiveLinkProps extends LinkProps{
  children: ReactElement;
  activeClassName: string;
}

export function ActiveLink({ children, activeClassName, ...rest }: ActiveLinkProps){
  const { asPath } = useRouter(); 

  const className = asPath === rest.href ? activeClassName : '';
  //Se a rota/pagina que estamos acessando for igual ao link que ele clicou entao ativamos o classname

  return(
    <Link {...rest} legacyBehavior>
      {cloneElement(children, {
        className
      })}
    </Link>
  )
}

    //alert('testando paginação', pageNumber)

    const prismic = getPrismicClient();

    const response = await prismic.query([

    ], {
      orderings: '[document.last_publication_date desc]', //Ordenar pelo mais recente
      fetch: ['post.title', 'post.description', 'post.cover'],
      pageSize: 3,// PAGINAÇÃO
      page: String(pageNumber)  // PAGINAÇÃO
    })

    return response;
  }


  // PAGINAÇÃO
  async function navigatePage(pageNumber: number){    
    const response = await reqPost(pageNumber);

    if(response.results.length === 0){
    return;      
    }


  // PAGINAÇÃO - utiliza esses dados das postagens
    const getPosts = response.results.map( post => {

        slug: post.uid,

        description: post.data.description.find(content => content.type === 'paragraph')?.text ?? '',
        cover: post.data.cover.url,
        updatedAt: new Date(post.last_publication_date).toLocaleDateString('pt-BR', {
          day: '2-digit',
          month: 'long',
          year: 'numeric'
        })
      }
    })

    setCurrentPage(pageNumber);   // PAGINAÇÃO -pagina atual

    setPosts(getPosts);  // PAGINAÇÃO

  }

              <time>{post.updatedAt}</time>
              <p>{post.description}</p>
            </a>
          </Link>
        ))}

         <div className={styles.buttonNavigate}>

          {/* // PAGINAÇÃO - Vai exibir botão de voltar se >=2 */}

              <div>
                <button onClick={ () => navigatePage(1) }>

                </button>
              </div>
            )}

{/* // PAGINAÇÃO - Vai exibir botão de avançar  */}
            { Number(currentPage) < Number(totalPage) && (
